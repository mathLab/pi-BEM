18// which needs, at construction time,
19// the specification of the number of
20// components.

22// For the exact solution the number

190// A boundary element method
191// triangulation is basically the
192// same as a (dim-1) dimensional
193// triangulation, with the difference
194// that the vertices belong to a
195// (dim) dimensional space.
196//
197// Some of the mesh formats supported
198// in deal.II use by default three
199// dimensional points to describe
200// meshes. These are the formats
201// which are compatible with the
202// boundary element method
203// capabilities of deal.II. In
204// particular we can use either UCD
205// or GMSH formats. In both cases, we
206// have to be particularly careful
207// with the orientation of the mesh,
208// because, unlike in the standard
209// finite element case, no reordering
210// or compatibility check is
211// performed here. All meshes are
212// considered as oriented, because
213// they are embedded in a higher
214// dimensional space. (See the
215// documentation of the GridIn and of
216// the Triangulation for further
217// details on orientation of cells in
218// a triangulation.) In our case, the
219// normals to the mesh are external
220// to both the circle in 2d or the
221// sphere in 3d.
222//
223// The other detail that is required
224// for appropriate refinement of the
225// boundary element mesh, is an
226// accurate description of the
227// manifold that the mesh is
228// approximating. We already saw this
229// several times for the boundary of
230// standard finite element meshes
231// (for example in step-5 and
232// step-6), and here the principle
233// and usage is the same, except that
234// the HyperBallBoundary class takes
235// an additional template parameter
236// that specifies the embedding space
237// dimension. The function object
238// still has to be static to live at
239// least as long as the triangulation
240// object to which it is attached.
241
242template <int dim>
243void
244ComputationalDomain<dim>::read_domain()
245{
246 std::ifstream in;
247 in.open(input_grid_name + "." + input_grid_format);
248 GridIn<dim - 1, dim> gi;
249 gi.attach_triangulation(tria);
250 if (input_grid_format == "vtk")
251 gi.read_vtk(in);
252 else if (input_grid_format == "msh")
253 gi.read_msh(in);
254 else if (input_grid_format == "inp")
255 gi.read_ucd(in, true);
256 else
257 Assert(false, ExcNotImplemented());
258
259 // GridTools::copy_material_to_manifold_id(tria);
260
261 if (input_grid_name == "../grids/coarse_sphere" ||
262 input_grid_name == "../grids/coarse_sphere_double_nodes" ||
263 input_grid_name == "../grids/circle")
264 {
265 manifold = new SphericalManifold<dim - 1, dim>;
266 tria.set_all_manifold_ids(0);
267 tria.set_manifold(0, *manifold);
268 used_spherical_manifold = true;
269 }

556template <int dim>
557void
558ComputationalDomain<dim>::refine_and_resize(const unsigned int refinement_level)
559{
560 pcout << "Refining and resizing mesh as required" << std::endl;
561
562
563 double max_tol = 0;
564 if (use_cad_surface_and_curves)
565 {
566 pcout << "Color Files" << endl;
567 unsigned int ii = 1;
568 bool go_on = true;
569 while (go_on == true)
570 {
571 std::string color_filename = (input_cad_path + "Color_" +
572 Utilities::int_to_string(ii) + ".iges");
573 ifstream f(color_filename);
574 if (f.good())
575 {
576 pcout << ii << "-th file exists" << endl;
577 TopoDS_Shape surface =
578 OpenCASCADE::read_IGES(color_filename, 1e-3);
579 cad_surfaces.push_back(surface);
580 }
581 else
582 go_on = false;
583 ii++;
584 }
585
586 pcout << "Edge Files" << endl;
587 ii = 1;
588 go_on = true;
589 while (go_on == true)
590 {
591 std::string edge_filename = (input_cad_path + "Curve_" +
592 Utilities::int_to_string(ii) + ".iges");
593 ifstream f(edge_filename);
594 if (f.good())
595 {
596 pcout << ii << "-th file exists" << endl;
597 TopoDS_Shape curve = OpenCASCADE::read_IGES(edge_filename, 1e-3);
598 cad_curves.push_back(curve);
599 }
600 else
601 go_on = false;
602 ii++;
603 }
604
605
606 for (unsigned int i = 0; i < cad_surfaces.size(); ++i)
607 {
608 pcout << i << endl;
609 max_tol =
610 fmax(max_tol, OpenCASCADE::get_shape_tolerance(cad_surfaces[i]));
611 pcout << max_tol << endl;
612 }
613 for (unsigned int i = 0; i < cad_curves.size(); ++i)
614 {
615 pcout << i + cad_surfaces.size() << endl;
616 max_tol =
617 fmax(max_tol, OpenCASCADE::get_shape_tolerance(cad_curves[i]));
618 pcout << max_tol << endl;
619 }
620
621 const double tolerance = cad_to_projectors_tolerance_ratio * max_tol;
622
623
624
625 pcout << "Used tolerance is: " << tolerance << endl;
626 for (unsigned int i = 0; i < cad_surfaces.size(); ++i)
627 {
628 normal_to_mesh_projectors.push_back(
629 std::make_shared<OpenCASCADE::NormalToMeshProjectionManifold<2, 3>>(
630 cad_surfaces[i], tolerance));
631 }
632 // static OpenCASCADE::DirectionalProjectionBoundary<2,3>
633 // directional_projector_lat(cad_surfaces[0],
634 // Point<3>(0.0,1.0,0.0), tolerance);
635 // static OpenCASCADE::NormalProjectionBoundary<2,3>
636 // normal_projector_lat(cad_surfaces[0], tolerance);
637
638 for (unsigned int i = 0; i < cad_curves.size(); ++i)
639 {
640 line_projectors.push_back(
641 std::make_shared<
642 OpenCASCADE::ArclengthProjectionLineManifold<2, 3>>(cad_curves[i],
643 tolerance));
644 }
645
646 for (unsigned int i = 0; i < cad_surfaces.size(); ++i)
647 {
648 tria.set_manifold(1 + i, *normal_to_mesh_projectors[i]);
649 }
650
651 for (unsigned int i = 0; i < cad_curves.size(); ++i)
652 {
653 tria.set_manifold(11 + i, *line_projectors[i]);
654 }
655 }
656
657
658 unsigned int refinedCellCounter = 1;
659 unsigned int cycles_counter = 0;
660 // we repeat the aspect ratio refinement cycle until no cell has been
661 // flagged for refinement, or until we reach a maximum of 10 cycles
662 while ((refinedCellCounter) && (cycles_counter < 10))
663 {
664 // the refined cells counter is zeroed at the start of each cycle
665 refinedCellCounter = 0;
666 // we loop on the all the triangulation active cells
667 Triangulation<2, 3>::active_cell_iterator cell = tria.begin_active();
