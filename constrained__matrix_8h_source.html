138void
139ConstrainedOperator<VEC, MATRIX>::distribute_rhs(VEC &rhs) const
140{
141 for (auto i : rhs.locally_owned_elements())
142 if ((constraints.is_constrained(i)) &&
143 (rhs.locally_owned_elements().is_element(i)))
144 rhs(i) = constraints.get_inhomogeneity(i);
