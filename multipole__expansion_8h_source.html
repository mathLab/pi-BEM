123 static FullMatrix<double>
124 A_n_m_Matrix(unsigned int dim)
125 {
126 FullMatrix<double> A_n_m(dim + 1, dim + 1);
127 for (unsigned int n = 0; n < dim + 1; n++)
128
129 {
130 for (unsigned int m = 0; m < n + 1; m++)
131
132 {
133 double f1 = 1.;
134 double f2 = 1.;
135
136 for (int ii = n - m; ii > 0; ii--)
137 f1 *= ii;
138
139 for (int ii = n + m; ii > 0; ii--)
140 f2 *= (ii);
141
142 A_n_m(n, m) = pow(-1., double(n)) / sqrt(f1 * f2);
143 }
144 }
145
146 return A_n_m;
147 }
